課題名           : flood_fill
提出ファイル      : flood_fill.c
使用可能関数      : -
--------------------------------------------------------------------------------

`char **` を2次元文字列配列として、`t_point` を配列の寸法として、`t_point` を開始点として受け取る関数を書いてください。

与えられた 'begin' `t_point` から開始して、この関数はゾーン内の文字を 'F' に置き換えることで、ゾーン全体を満たします。ゾーンは同じ文字の群で、他の文字または配列の境界によって水平および垂直に区切られています。

`flood_fill` 関数は対角線方向には満たしません。

`flood_fill` 関数は以下のようにプロトタイプされます：
  void  flood_fill(char **tab, t_point size, t_point begin);

`t_point` 構造体は以下のようにプロトタイプされます（flood_fill.c に入力してください）：

  typedef struct  s_point
  {
    int           x;
    int           y;
  }               t_point;

実行例:

$> cat test.c
#include <stdlib.h>
#include <stdio.h>

char** make_area(char** zone, t_point size)
{
	char** new;

	new = malloc(sizeof(char*) * size.y);
	for (int i = 0; i < size.y; ++i)
	{
		new[i] = malloc(size.x + 1);
		for (int j = 0; j < size.x; ++j)
			new[i][j] = zone[i][j];
		new[i][size.x] = '\0';
	}

	return new;
}

int main(void)
{
	t_point size = {8, 5};
	char *zone[] = {
		"11111111",
		"10001001",
		"10010001",
		"10110001",
		"11100001",
	};

	char**  area = make_area(zone, size);
	for (int i = 0; i < size.y; ++i)
		printf("%s\n", area[i]);
	printf("\n");

	t_point begin = {7, 4};
	flood_fill(area, size, begin);
	for (int i = 0; i < size.y; ++i)
		printf("%s\n", area[i]);
	return (0);
}

$> gcc flood_fill.c test.c -o test; ./test
